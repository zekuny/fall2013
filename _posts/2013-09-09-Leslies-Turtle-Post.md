import turtle
bri=turtle.Turtle()

turtle.pencolor('violet')
bri.pencolor('violet')
bri.setposition(-30,50)
bri.setposition(-30,100)
bri.setposition(0,0)
bri.setposition(30,50)
bri.setposition(30,100)
bri.setposition(0,0)
bri.setposition(30,-50)
bri.setposition(30,-100)

bri.setposition(-30,-100)
bri.setposition(0,0)
